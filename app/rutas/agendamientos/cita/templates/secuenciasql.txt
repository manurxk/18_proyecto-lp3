CREATE TABLE consulta (
    id INT AUTO_INCREMENT PRIMARY KEY,
    nombrepaciente VARCHAR(100), -- Ajusta el tamaño según sea necesario
    motivoconsulta VARCHAR(255), -- Ajusta el tamaño según sea necesario
    medico VARCHAR(100), -- Nombre del médico
    fecha DATE,          -- Fecha de la consulta
    hora TIME            -- Hora de la consulta
);
